      def PreProcessing(df, sigidx, dataidx):
          
          data_sig = list(df.iloc[:,sigidx])
          del_list = []
          
          no_forium_idx = []
          no_forium = []
          no_forium_sig = []
          data_mean_list = []
          
          #data의 sig코드가 json파일의 sig코드에 없으면 data에서 해당 행을 삭제
          for i,sig in enumerate(data_sig):
              if sig in json_sig:
                  pass
              else:
                  del_list.append(i)
          drop_data = df.drop(del_list, axis=0)
          drop_data_sig = list(drop_data.iloc[:,sigidx])
          
          
          #json파일의 sig코드중에 drop data에 없어 forium에 표시가 안되는 지역 추출
          for i,sig in enumerate(json_sig):
              if sig in drop_data_sig:
                  pass
              else:
                  no_forium_idx.append(i)
          for i in no_forium_idx:
              #buf = str(json_name[i]) + ":" + str(json_sig[i])
              no_forium_sig.append(json_sig[i])
              #no_forium.append(buf)
              
          
          #forium에서 결측치 없이 표현 할 수 있도록 dataframe 재구성
          data_mean = df.iloc[:,dataidx].mean() + 10
          print(data_mean)
          #for_length = len(no_forium_sig)
          for i in range(len(no_forium_sig)):
              data_mean_list.append(data_mean)
          sig_name = df.columns[sigidx]
          data_name = df.columns[dataidx]
          new_data = pd.DataFrame({sig_name:no_forium_sig, data_name:data_mean_list})
          result_df = pd.concat([drop_data, new_data], ignore_index=True)
          
          
          
          return result_df
